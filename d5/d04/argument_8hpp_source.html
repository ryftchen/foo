82template <typename T>
83std::string represent(const T& val)
84{
85 if constexpr (std::is_same_v<T, bool>)
86 {
87 return val ? "true" : "false";
88 }
89 else if constexpr (std::is_convertible_v<T, std::string_view>)
90 {
91 return '"' + std::string{std::string_view{val}} + '"';
92 }
93 else if constexpr (isContainer<T>)
94 {
95 std::ostringstream out;
96 out << '{';
97 const auto size = val.size();
98 if (size > 1)
99 {
100 out << represent(*val.begin());
101 std::for_each(
102 std::next(val.begin()),
103 std::next(
104 val.begin(),
105 static_cast<typename T::iterator::difference_type>(
106 std::min<std::size_t>(size, representMaxContainerSize) - 1)),
107 [&out](const auto& v)
108 {
109 out << ' ' << represent(v);
110 });
111 if (size <= representMaxContainerSize)
112 {
113 out << ' ';
114 }
115 else
116 {
117 out << " ... ";
118 }
119 }
120 if (size > 0)
121 {
122 out << represent(*std::prev(val.end()));
123 }
124 out << '}';
125 return std::move(out).str();
126 }
127 else if constexpr (isStreamable<T>)
128 {
129 std::ostringstream out;
130 out << val;
131 return std::move(out).str();
132 }
133 else
134 {
135 return " not representable ";
136 }
137}
