34void Log::runLogger()
35{
36 State expectedState = State::init;
37 const auto checkIfExceptedFSMState = [this, &expectedState](const State state)
38 {
39 expectedState = state;
40 if (currentState() != expectedState)
41 {
42 throw std::logic_error("Abnormal logger state.");
43 }
44 };
45
46launchPoint:
47 try
48 {
49 checkIfExceptedFSMState(State::init);
50 processEvent(OpenFile());
51
52 checkIfExceptedFSMState(State::idle);
53 processEvent(GoLogging());
54
55 checkIfExceptedFSMState(State::work);
56 while (isLogging.load())
57 {
58 std::unique_lock<std::mutex> lock(mtx);
59 cv.wait(
60 lock,
61 [this]()
62 {
63 return (!isLogging.load() || !logQueue.empty() || rollbackRequest.load());
64 });
65
66 if (rollbackRequest.load())
67 {
68 break;
69 }
70 handleLogQueue();
71 }
72
73 if (rollbackRequest.load())
74 {
75 processEvent(Relaunch());
76 goto launchPoint; // NOLINT (hicpp-avoid-goto)
77 }
78 processEvent(CloseFile());
79
80 checkIfExceptedFSMState(State::idle);
81 processEvent(NoLogging());
82
83 checkIfExceptedFSMState(State::done);
84 }
85 catch (const std::exception& error)
86 {
87 LOG_ERR << error.what() << " Expected logger state: " << expectedState
88 << ", current logger state: " << State(currentState()) << '.';
89 processEvent(Standby());
90
91 checkIfExceptedFSMState(State::hold);
92 if (awaitNotification4Rollback())
93 {
94 goto launchPoint; // NOLINT (hicpp-avoid-goto)
95 }
96 }
97}

99void Log::waitToStart()
100{
101 while (!((currentState() == State::idle) && !rollbackRequest.load()))
102 {
103 if ((currentState() == State::hold) && !rollbackRequest.load())
104 {
105 LOG_ERR << "The logger did not initialize successfully...";
106 return;
107 }
108 utility::time::millisecondLevelSleep(intervalOfWaitLogger);
109 }
110
111 if (std::unique_lock<std::mutex> lock(mtx); true)
112 {
113 isLogging.store(true);
114
115 lock.unlock();
116 cv.notify_one();
117 }
118
119 utility::time::BlockingTimer expiryTimer;
120 std::uint16_t waitCount = 0;
121 expiryTimer.set(
122 [this, &expiryTimer, &waitCount]()
123 {
124 if ((currentState() == State::work) && !rollbackRequest.load())
125 {
126 expiryTimer.reset();
127 }
128 else
129 {
130 ++waitCount;
131 }
132
133 if (maxTimesOfWaitLogger == waitCount)
134 {
135 LOG_ERR << "The logger did not start properly...";
136 expiryTimer.reset();
137 }
138 },
139 intervalOfWaitLogger);
140}
