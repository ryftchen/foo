93int Console::commandExecutor(const std::string& command)
94{
95 std::vector<std::string> inputs;
96 std::istringstream is(command);
97 std::copy(std::istream_iterator<std::string>(is), std::istream_iterator<std::string>(), std::back_inserter(inputs));
98
99 if (inputs.empty())
100 {
101 return RetCode::success;
102 }
103
104 const auto iterator = impl->regMap.find(inputs.at(0));
105 if (std::cend(impl->regMap) == iterator)
106 {
107 throw std::logic_error("The console command '" + inputs.at(0) + "' could not be found.");
108 }
109
110 return RetCode(static_cast<int>(std::get<0>(iterator->second)(inputs)));
111}

113int Console::fileExecutor(const std::string& filename)
114{
115 std::ifstream input(filename);
116 if (!input)

118 throw std::runtime_error("Could not find the batch file to run.");
119 }
120
121 std::string command;

133 {
134 return RetCode(result);
135 }

138 }
139

216char* Console::getCommandIterator(const char* text, int state)
217{
218 static Impl::RegisteredCommand::iterator iterator;

229
