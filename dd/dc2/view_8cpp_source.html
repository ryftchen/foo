36int tlvEncode(char* buf, int& len, const TLVValue& val)
37{
38 if (nullptr == buf)
39 {
40 return -1;
41 }
42
43 constexpr int offset = sizeof(int) + sizeof(int);
44 int sum = 0;
45
46 Packet enc(buf, len);
47 enc.write<int>(TLVType::header);
48 enc.write<int>(sum);
49
50 enc.write<int>(TLVType::stop);
51 enc.write<int>(sizeof(bool));
52 enc.write<bool>(val.stopTag);
53 sum += (offset + sizeof(bool));
54
55 const auto tryToFillShmId = [&](const TLVType type, int TLVValue::*shmId)
56 {
57 if (invalidShmId != val.*shmId)
58 {
59 enc.write<int>(type);
60 enc.write<int>(sizeof(int));
61 enc.write<int>(val.*shmId);
62 sum += (offset + sizeof(int));
63 }
64 };
65 tryToFillShmId(TLVType::bash, &TLVValue::bashShmId);
66 tryToFillShmId(TLVType::journal, &TLVValue::logShmId);
67 tryToFillShmId(TLVType::monitor, &TLVValue::statusShmId);
68
69 *reinterpret_cast<int*>(buf + sizeof(int)) = ::htonl(sum);
70 len = offset + sum;
71
72 return 0;
73}

80int tlvDecode(char* buf, const int len, TLVValue& val)
81{
82 if (nullptr == buf)
83 {
84 return -1;
85 }
86
87 Packet dec(buf, len);
88 int type = 0, length = 0, sum = 0;
89
90 dec.read<int>(&type);
91 if (TLVType::header != type)
92 {
93 return -1;
94 }
95 dec.read<int>(&sum);
96
97 constexpr int offset = sizeof(int) + sizeof(int);
98 while (sum > 0)
99 {
100 dec.read<int>(&type);
101 dec.read<int>(&length);
102 switch (type)
103 {
104 case TLVType::stop:
105 dec.read<bool>(&val.stopTag);
106 sum -= (offset + sizeof(bool));
107 break;
108 case TLVType::bash:
109 dec.read<int>(&val.bashShmId);
110 sum -= (offset + sizeof(int));
111 break;
112 case TLVType::journal:
113 dec.read<int>(&val.logShmId);
114 sum -= (offset + sizeof(int));
115 break;
116 case TLVType::monitor:
117 dec.read<int>(&val.statusShmId);
118 sum -= (offset + sizeof(int));
119 break;
120 default:
121 break;
122 }
123 }
124
125 return 0;
126}

193void View::runViewer()
194{
195 State expectedState = State::init;
196 const auto checkIfExceptedFSMState = [this, &expectedState](const State state)
197 {
198 expectedState = state;
199 if (currentState() != expectedState)
200 {
201 throw std::logic_error("Abnormal viewer state.");
202 }
203 };
204
205launchPoint:
206 try
207 {
208 checkIfExceptedFSMState(State::init);
209 processEvent(CreateServer());
210
211 checkIfExceptedFSMState(State::idle);
212 processEvent(GoViewing());
213
214 checkIfExceptedFSMState(State::work);
215 while (isViewing.load())
216 {
217 std::unique_lock<std::mutex> lock(mtx);
218 cv.wait(
219 lock,
220 [this]()
221 {
222 return (!isViewing.load() || rollbackRequest.load());
223 });
224
225 if (rollbackRequest.load())
226 {
227 break;
228 }
229 }
230
231 if (rollbackRequest.load())
232 {
233 processEvent(Relaunch());
234 goto launchPoint; // NOLINT (hicpp-avoid-goto)
235 }
236 processEvent(DestroyServer());
237
238 checkIfExceptedFSMState(State::idle);
239 processEvent(NoViewing());
240
241 checkIfExceptedFSMState(State::done);
242 }
243 catch (const std::exception& error)
244 {
245 LOG_ERR << error.what() << " Expected viewer state: " << expectedState
246 << ", current viewer state: " << State(currentState()) << '.';
247 processEvent(Standby());
248
249 checkIfExceptedFSMState(State::hold);
250 if (awaitNotification4Rollback())
251 {
252 goto launchPoint; // NOLINT (hicpp-avoid-goto)
253 }
254 }

257void View::waitToStart()
258{
259 while (!((currentState() == State::idle) && !rollbackRequest.load()))
260 {
261 if ((currentState() == State::hold) && !rollbackRequest.load())
262 {
263 LOG_ERR << "The viewer did not initialize successfully...";

266 utility::time::millisecondLevelSleep(intervalOfWaitViewer);

268
269 if (std::unique_lock<std::mutex> lock(mtx); true)
270 {
271 isViewing.store(true);

499int View::fillSharedMemory(const std::string& contents)
500{
501 const int shmId = ::shmget(
502 static_cast<::key_t>(0),
503 sizeof(SharedMemory),
504 IPC_CREAT | IPC_EXCL | S_IRUSR | S_IWUSR | S_IRGRP | S_IWGRP | S_IROTH | S_IWOTH);
505 if (-1 == shmId)
506 {
507 throw std::runtime_error("Failed to create shared memory.");
508 }
509 void* const shm = ::shmat(shmId, nullptr, 0);
510 if (nullptr == shm)
511 {
512 throw std::runtime_error("Failed to attach shared memory.");
513 }
514
515 auto* const shrMem = reinterpret_cast<SharedMemory*>(shm);
516 shrMem->signal.store(false);
517 for (;;)
518 {
519 if (!shrMem->signal.load())
520 {
521 std::memset(shrMem->buffer, 0, sizeof(shrMem->buffer));
522 std::memcpy(shrMem->buffer, contents.c_str(), std::min(sizeof(shrMem->buffer), contents.length()));
523 shrMem->signal.store(true);
524 break;
525 }
526 utility::time::millisecondLevelSleep(1);
527 }
528 ::shmdt(shm);
529
530 return shmId;
531}

635{

637 tcpServer->onNewConnection = [this](utility::socket::TCPSocket* const newSocket)
638 {
639 newSocket->onMessageReceived = [this, newSocket](const std::string& message)
640 {
641 try
642 {
643 if (0 == message.length())
644 {
645 return;
646 }
647
648 char buffer[maxMsgLength] = {'\0'};
649 const auto msg = utility::common::base64Decode(message);
650 if ("stop" == msg)
651 {
652 buildTLVPacket2Stop(buffer);
653 newSocket->toSend(buffer, sizeof(buffer));
654 newSocket->setNonBlocking();
655 return;
656 }
657
658 std::vector<std::string> args;
