352template <typename... Args>
353void Log::flush(
354 const OutputLevel level,
355 const std::string& codeFile,
356 const std::uint32_t codeLine,
357 const std::string& format,
358 Args&&... args)
359{
360 const auto outputFormatter = [&](const std::string_view& prefix)
361 {
362 std::string validFormat = format;
363 validFormat.erase(
364 std::remove_if(
365 std::begin(validFormat),
366 std::end(validFormat),
367 [l = std::locale{}](const auto c)
368 {
369 return ((' ' != c) && std::isspace(c, l));
370 }),
371 std::end(validFormat));
372 std::string output = std::string{prefix} + ":[" + utility::time::getCurrentSystemTime() + "]:["
373 + codeFile.substr(codeFile.find("foo/") + 4, codeFile.length()) + '#' + std::to_string(codeLine)
374 + "]: " + utility::common::formatString(validFormat.c_str(), std::forward<Args>(args)...);
375 return output;
376 };
377
378 if (!((State::work == currentState()) && !rollbackRequest.load()))
379 {
380 std::string output = outputFormatter(unknownLevelPrefix);
381 std::cerr << changeToLogStyle(output) << std::endl;
382 return;
383 }
384
385 if (std::unique_lock<std::mutex> lock(mtx); true)
386 {
387 if (level >= minimumLevel)
388 {
389 std::string_view prefix;
390 switch (level)
391 {
392 case OutputLevel::debug:
393 prefix = debugLevelPrefix;
394 break;
395 case OutputLevel::info:
396 prefix = infoLevelPrefix;
397 break;
398 case OutputLevel::warning:
399 prefix = warnLevelPrefix;
400 break;
401 case OutputLevel::error:
402 prefix = errorLevelPrefix;
403 break;
404 default:
405 break;
406 }
407
408 std::string output = outputFormatter(prefix);
409 logQueue.push(std::move(output));
410 lock.unlock();
411 cv.notify_one();
412 }
413 }
414}
