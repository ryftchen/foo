505{
506 std::vector<std::string> arguments;
507 for (const auto& arg : rawArguments)
508 {
509 const auto argumentStartsWithPrefixChars = [this](const std::string& str)
510 {
511 if (!str.empty())
512 {
513 const auto legalPrefix = [this](const char c)
514 {
515 return (prefixChars.find(c) != std::string::npos);
516 };
517
518 const auto windowsStyle = legalPrefix('/');
519 if (windowsStyle)
520 {
521 if (legalPrefix(str.at(0)))
522 {
523 return true;
524 }
525 }
526 else
527 {
528 if (str.size() > 1)
529 {
530 return (legalPrefix(str.at(0)) && legalPrefix(str.at(1)));
531 }
532 }
533 }
534 return false;
535 };
536
537 const auto assignCharPos = arg.find_first_of(assignChars);
538 if ((argumentMap.find(arg) == argumentMap.cend()) && argumentStartsWithPrefixChars(arg)
539 && (std::string::npos != assignCharPos))

541 const std::string optName = arg.substr(0, assignCharPos);
542 if (argumentMap.find(optName) != argumentMap.cend())

544 arguments.emplace_back(optName);
545 arguments.emplace_back(arg.substr(assignCharPos + 1));

548 }
549 arguments.emplace_back(arg);

555{
