508 {{Rastrigin::range1, Rastrigin::range2, Rastrigin::funcDescr}, Rastrigin{}},

517 std::cout << prefix << Rastrigin::funcDescr << std::endl;
518 },
519 },
520 function);
521 };
522 const auto calcFunc = [&targets](const optimal::Function& function, const optimal::FuncRange<double, double>& range)
523 {
524 assert(range.range1 < range.range2);
525 auto* const threads = command::getPublicThreadPool().newElement(std::min(
526 static_cast<std::uint32_t>(getBit<OptimalMethod>().count()),
527 static_cast<std::uint32_t>(Bottom<OptimalMethod>::value)));
528 const auto optimalFunctor =
529 [threads, &function, &range](
530 const std::string& threadName, void (*methodPtr)(const optimal::Function&, const double, const double))
531 {
532 threads->enqueue(threadName, methodPtr, std::ref(function), range.range1, range.range2);
533 };
534 const auto name = utility::currying::curry(getTaskNameCurried(), "o");
535
536 using optimal::OptimalSolution;
537 using utility::common::operator""_bkdrHash;
538 for (std::uint8_t i = 0; i < Bottom<OptimalMethod>::value; ++i)

955void runSortTasks(const std::vector<std::string>& targets)
956{
957 if (getBit<SortMethod>().none())
958 {
959 return;
960 }
961
962 APP_ALGO_PRINT_TASK_BEGIN_TITLE(Category::sort);
963 using sort::SortSolution;
964 using sort::TargetBuilder;
965 using sort::input::arrayLength;
966 using sort::input::arrayRange1;
967 using sort::input::arrayRange2;
968 using utility::common::operator""_bkdrHash;
969 static_assert((arrayRange1 < arrayRange2) && (arrayLength > 0));
970
971 auto* const threads = command::getPublicThreadPool().newElement(std::min(
972 static_cast<std::uint32_t>(getBit<SortMethod>().count()),
973 static_cast<std::uint32_t>(Bottom<SortMethod>::value)));
974 const auto builder = std::make_shared<TargetBuilder<int>>(arrayLength, arrayRange1, arrayRange2);
975 const auto sortFunctor =
976 [threads, builder](const std::string& threadName, void (*methodPtr)(const int* const, const std::uint32_t))
977 {
978 threads->enqueue(threadName, methodPtr, builder->getRandomArray().get(), builder->getLength());
979 };
980 const auto name = utility::currying::curry(getTaskNameCurried(), "S");
981
982 for (std::uint8_t i = 0; i < Bottom<SortMethod>::value; ++i)
983 {
984 if (!getBit<SortMethod>().test(SortMethod(i)))
985 {
986 continue;
987 }
988
989 const std::string targetMethod = targets.at(i);
990 switch (utility::common::bkdrHash(targetMethod.data()))
991 {
992 case "bub"_bkdrHash:
993 sortFunctor(name(targetMethod), &SortSolution::bubbleMethod);
994 break;
995 case "sel"_bkdrHash:
996 sortFunctor(name(targetMethod), &SortSolution::selectionMethod);
997 break;
998 case "ins"_bkdrHash:
999 sortFunctor(name(targetMethod), &SortSolution::insertionMethod);
1000 break;
1001 case "she"_bkdrHash:
1002 sortFunctor(name(targetMethod), &SortSolution::shellMethod);
1003 break;
1004 case "mer"_bkdrHash:
1005 sortFunctor(name(targetMethod), &SortSolution::mergeMethod);
1006 break;
1007 case "qui"_bkdrHash:
1008 sortFunctor(name(targetMethod), &SortSolution::quickMethod);
1009 break;
1010 case "hea"_bkdrHash:
1011 sortFunctor(name(targetMethod), &SortSolution::heapMethod);
1012 break;
1013 case "cou"_bkdrHash:
1014 sortFunctor(name(targetMethod), &SortSolution::countingMethod);
1015 break;
1016 case "buc"_bkdrHash:
1017 sortFunctor(name(targetMethod), &SortSolution::bucketMethod);
1018 break;
1019 case "rad"_bkdrHash:
1020 sortFunctor(name(targetMethod), &SortSolution::radixMethod);
1021 break;
1022 default:
1023 LOG_INF << "Execute to apply an unknown sort method.";
1024 break;
1025 }
1026 }
