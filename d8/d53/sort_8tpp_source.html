163template <class T>
164void Sort<T>::quickSortRecursive(T* const sortArray, const std::uint32_t begin, const std::uint32_t end)
165{
166 if (begin >= end)
167 {
168 return;
169 }
170 T pivot = sortArray[end];
171 std::uint32_t leftIndex = begin, rightIndex = end - 1;
172
173 while (leftIndex < rightIndex)
174 {
175 while ((leftIndex < rightIndex) && (sortArray[leftIndex] < pivot))
176 {
177 ++leftIndex;
178 }
179 while ((leftIndex < rightIndex) && (sortArray[rightIndex] >= pivot))
180 {
181 --rightIndex;
182 }
183 std::swap(sortArray[leftIndex], sortArray[rightIndex]);
184 }
185 if (sortArray[leftIndex] >= sortArray[end])
186 {
187 std::swap(sortArray[leftIndex], sortArray[end]);
188 }
189 else
190 {
191 ++leftIndex;
192 }
193
194 if (leftIndex)
195 {
196 quickSortRecursive(sortArray, begin, leftIndex - 1);
197 }
198 quickSortRecursive(sortArray, leftIndex + 1, end);

223{

225 while (child <= end)
226 {
