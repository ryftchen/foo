116void TCPSocket::toConnect(const std::string& ip, const std::uint16_t port, const std::function<void()> onConnected)
117{
118 ::addrinfo hints{}, *res = nullptr, *it = nullptr;
119 std::memset(&hints, 0, sizeof(hints));
120 hints.ai_family = AF_INET;
121 hints.ai_socktype = ::SOCK_STREAM;
122
123 int status = 0;
124 if ((status = ::getaddrinfo(ip.c_str(), nullptr, &hints, &res)) != 0)
125 {
126 throw std::runtime_error(
127 "Invalid address, status: " + std::string(::gai_strerror(status)) + ", errno: " + std::to_string(errno)
128 + '.');
129 }
130
131 for (it = res; nullptr != it; it = it->ai_next)
132 {
133 if (AF_INET == it->ai_family)
134 {
135 std::memcpy(static_cast<void*>(&sockAddr), static_cast<void*>(it->ai_addr), sizeof(::sockaddr_in));
136 break;
137 }
138 }
139 ::freeaddrinfo(res);
140
141 sockAddr.sin_family = AF_INET;
142 sockAddr.sin_port = ::htons(port);
143 sockAddr.sin_addr.s_addr = static_cast<std::uint32_t>(sockAddr.sin_addr.s_addr);
144
145 setBlocking();
146 if (::connect(sock, reinterpret_cast<const ::sockaddr*>(&sockAddr), sizeof(::sockaddr_in)) == -1)
147 {
148 throw std::runtime_error("Could not connect to the socket, errno: " + std::to_string(errno) + '.');
149 }
150
151 onConnected();

258void TCPServer::toAccept(const TCPServer* const server)
259{
260 ::sockaddr_in newSocketInfo{};
261 ::socklen_t newSocketInfoLength = sizeof(newSocketInfo);
262
263 int newSock = 0;
264 for (;;)
265 {
266 if ((newSock = ::accept(server->sock, reinterpret_cast<::sockaddr*>(&newSocketInfo), &newSocketInfoLength))
267 == -1)
268 {
269 if ((EBADF == errno) || (EINVAL == errno))
270 {
271 return;
272 }
273
274 throw std::runtime_error("Error while accepting a new connection, errno: " + std::to_string(errno) + '.');
275 }
276
277 TCPSocket* const newSocket = new TCPSocket(newSock);
278 newSocket->autoRelease.store(true);
279 newSocket->setAddress(newSocketInfo);
280
281 server->onNewConnection(newSocket);
282 newSocket->toReceive(true);
283 }
