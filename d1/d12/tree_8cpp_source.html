735Node* splayTreeSplay(SplayTree tree, const Type key)
736{
737 Node n{}, *l = nullptr, *r = nullptr, *c = nullptr;
738 if (nullptr == tree)
739 {
740 return tree;
741 }
742
743 n.left = n.right = nullptr;
744 l = r = &n;
745 for (;;)
746 {
747 if (key < tree->key)
748 {
749 if (nullptr == tree->left)
750 {
751 break;
752 }
753 if (key < tree->left->key)
754 {
755 c = tree->left;
756 tree->left = c->right;
757 c->right = tree;
758 tree = c;
759 if (nullptr == tree->left)
760 {
761 break;
762 }
763 }
764 r->left = tree;
765 r = tree;
766 tree = tree->left;
767 }
768 else if (key > tree->key)
769 {
770 if (nullptr == tree->right)
771 {
772 break;

775 {
776 c = tree->right;
777 tree->right = c->left;
778 c->left = tree;
779 tree = c;
780 if (nullptr == tree->right)
781 {
782 break;
783 }
784 }
785 l->right = tree;
786 l = tree;
787 tree = tree->right;
788 }
789 else
790 {
791 break;
792 }
793 }

795 l->right = tree->left;
796 r->left = tree->right;
797 tree->left = n.right;
798 tree->right = n.left;
