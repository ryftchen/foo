121void runLinearTasks(const std::vector<std::string>& targets)
122{
123 if (getBit<LinearInstance>().none())
124 {
125 return;
126 }
127
128 APP_DS_PRINT_TASK_BEGIN_TITLE(Category::linear);
129 using linear::LinearStructure;
130 using utility::common::operator""_bkdrHash;
131
132 auto* const threads = command::getPublicThreadPool().newElement(std::min(
133 static_cast<std::uint32_t>(getBit<LinearInstance>().count()),
134 static_cast<std::uint32_t>(Bottom<LinearInstance>::value)));
135 const auto linearFunctor = [threads](const std::string& threadName, void (*instancePtr)())
136 {
137 threads->enqueue(threadName, instancePtr);
138 };
139 const auto name = utility::currying::curry(getTaskNameCurried(), "l");
140
141 std::cout << "\r\nInstances of the linear structure:" << std::endl;
142 for (std::uint8_t i = 0; i < Bottom<LinearInstance>::value; ++i)
143 {
144 if (!getBit<LinearInstance>().test(LinearInstance(i)))
145 {
146 continue;
147 }
148
149 const std::string targetInstance = targets.at(i);
150 switch (utility::common::bkdrHash(targetInstance.data()))
151 {
152 case "lin"_bkdrHash:
153 linearFunctor(name(targetInstance), &LinearStructure::linkedListInstance);
154 break;
155 case "sta"_bkdrHash:
156 linearFunctor(name(targetInstance), &LinearStructure::stackInstance);
157 break;
158 case "que"_bkdrHash:
159 linearFunctor(name(targetInstance), &LinearStructure::queueInstance);
160 break;
161 default:
162 LOG_INF << "Execute to apply an unknown linear instance.";
163 break;
164 }
165 }
166
167 command::getPublicThreadPool().deleteElement(threads);
168 APP_DS_PRINT_TASK_END_TITLE(Category::linear);
169}
