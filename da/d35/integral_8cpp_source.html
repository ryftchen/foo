25double trapezoid(const Expression& expr, const double left, const double height, const std::uint32_t step)
26{
27 double sum = 0.0, x = left;
28 const double delta = height / step;
29 for (std::uint32_t i = 0; i < step; ++i)
30 {
31 const double area = (expr(x) + expr(x + delta)) * delta / 2.0; // S=(a+b)*h/2
32 sum += area;
33 x += delta;
34 }
35 return sum;
36}

38double Trapezoidal::operator()(double lower, double upper, const double eps) const
39{
40 const int sign = getSign(lower, upper);
41 const std::uint32_t minStep = std::pow(2, 3);
42 const double height = upper - lower;
43 double sum = 0.0, s1 = 0.0, s2 = 0.0;
44 std::uint32_t n = 1;
45
46 do
47 {
48 sum = trapezoid(expr, lower, height, n);
49 s1 = s2;
50 s2 = sum;
51 n *= 2;
52 }
53 while ((std::fabs(s1 - s2) > eps) || (n < minStep));
54 sum = s2 * sign;

79double Simpson::compositeSimpsonOneThird(const double left, const double right, const std::uint32_t step) const
80{
81 const double stepLength = (right - left) / step;
82 double sum = 0.0;
83 for (std::uint32_t i = 0; i < step; ++i)
84 {
85 // I≈(b-a)/6[Y0+Y2n+4(Y1+...+Y2n-1)+6(Y2+...+Y2n-2)]
86 sum += simpsonOneThird(left + i * stepLength, left + (i + 1) * stepLength);
87 }
88 return sum;

96double Romberg::operator()(double lower, double upper, const double eps) const
97{
98 const int sign = getSign(lower, upper);
99 std::uint32_t k = 0;
100 double sum = 0.0;
101 const double height = upper - lower;
102 const auto trapezoidFunctor = std::bind(trapezoid, std::ref(expr), lower, height, std::placeholders::_1);
103 double t0 = trapezoidFunctor(std::pow(2, k));
104
105 k = 1;
106 double t1Zero = trapezoidFunctor(std::pow(2, k));
107 double t1 =
108 std::pow(4, k) / (std::pow(4, k) - 1) * trapezoidFunctor(std::pow(2, k + 1)) - 1.0 / std::pow(4, k) * t1Zero;
109
110 while (std::fabs(t1 - t0) > eps)
111 {
112 ++k;
113 t0 = t1;
114 t1Zero = trapezoidFunctor(std::pow(2, k));
115 for (std::uint32_t i = 1; i <= k; ++i)
116 {
117 t1 = std::pow(4, i) / (std::pow(4, i) - 1) * trapezoidFunctor(std::pow(2, i + 1))
118 - 1.0 / std::pow(4, i) * t1Zero;
119 }
120 }
121 sum = trapezoidFunctor(std::pow(2, k)) * sign;
122
123 return sum;
124}

126double Gauss::operator()(double lower, double upper, const double eps) const
127{
128 const int sign = getSign(lower, upper);
129 constexpr std::uint32_t gaussNodes = 5, gaussCoefficient = 2;
130 constexpr std::array<std::array<double, gaussCoefficient>, gaussNodes> gaussLegendreTable = {
131 {{-0.9061798459, +0.2369268851},
132 {-0.5384693101, +0.4786286705},
133 {+0.0000000000, +0.5688888889},
134 {+0.5384693101, +0.4786286705},
135 {+0.9061798459, +0.2369268851}}};
136 double sum = 0.0, s1 = 0.0, s2 = 0.0;
137 std::uint32_t n = 1;
138
139 do
140 {
141 sum = 0.0;
142 const double stepLength = (upper - lower) / n;
143 for (std::uint32_t i = 0; i < n; ++i)
144 {
145 const double left = lower + i * stepLength, right = left + stepLength;
146 for (std::uint32_t j = 0; j < gaussNodes; ++j)
147 {
148 // x=1/2[(a+b)+(b-a)t]
149 const double x = ((right - left) * gaussLegendreTable.at(j).at(0) + (left + right)) / 2.0,
150 polynomial = expr(x) * gaussLegendreTable.at(j).at(1) * (right - left) / 2.0;
151 sum += polynomial;
152 }
153 }
154 s1 = s2;
155 s2 = sum;
156 n *= 2;
157 }
158 while (std::fabs(s1 - s2) > eps);
159 sum = s2 * sign;
160
161 return sum;
162}

174double MonteCarlo::sampleFromUniformDistribution(const double lower, const double upper, const double eps) const
175{

177 std::vector<double> cache(n);

191double MonteCarlo::sampleFromNormalDistribution(const double lower, const double upper, const double eps) const
192{
193 const std::uint32_t n = std::max<std::uint32_t>((upper - lower) / eps, 1.0 / eps);
194 const double mu = (lower + upper) / 2.0, sigma = (upper - lower) / 6.0;
195 std::mt19937 engine(std::random_device{}());
196 std::uniform_real_distribution<double> dist(0.0, 1.0);
197 double sum = 0.0, x = 0.0;
198 for (std::uint32_t i = 0; i < n; ++i)
199 {
200 do
201 {
202 const double u1 = dist(engine), u2 = dist(engine), mag = sigma * std::sqrt(-2.0 * std::log(u1));
203 x = mag * std::sin(2.0 * std::numbers::pi * u2) + mu; // Box-Muller Transform
204 }
205 while ((x < lower) || (x > upper));
206 const double probabilityDensityFunction = (1.0 / std::sqrt(2.0 * std::numbers::pi * sigma * sigma))
207 * std::pow(M_E, (-(x - mu) * (x - mu)) / (2.0 * sigma * sigma));
208 sum += expr(x) / probabilityDensityFunction; // I≈1/N*[F(X1)/P(X1)+...+F(Xn)/P(Xn)]
209 }
210 sum /= n;
211
212 return sum;
213}
